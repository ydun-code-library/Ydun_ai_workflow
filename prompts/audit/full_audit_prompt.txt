## SEVERITY CLASSIFICATION WITH VERIFICATION

```python
def classify_severity(finding, verification_status):
    # Only classify if verification passed
    if verification_status != "VERIFIED":
        return "UNVERIFIED"  # Blocked from report
    
    if any([
        "exposed_password" in finding,
        "sql_injection" in finding,
        "security_vulnerability" in finding,
        "production_break" in finding
    ]):
        return "CRITICAL"
    
    elif any([
        finding.complexity > 15,
        "SOLID_violation" in finding,
        finding.coverage < 20,
        "data_leak" in finding
    ]):
        return "HIGH"
    
    elif any([
        finding.duplication > 20,
        "missing_documentation" in finding,
        finding.outdated_major_version
    ]):
        return "MEDIUM"
    
    elif any([
        "formatting" in finding,
        "naming_inconsistency" in finding
    ]):
        return "LOW"
    
    else:
        return "INFO"
```
